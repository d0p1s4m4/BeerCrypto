61static uint16_t
62MISTY1_FI(uint16_t in, uint16_t key)
63{
64 uint16_t d9;
65 uint16_t d7;
66
67 d9 = in >> 7;
68 d7 = in & 0x7F;
69 d9 = S9[d9] ^ d7;
70 d7 = S7[d7] ^ d9;
71 d7 = d7 & 0x7F;
72 d7 = d7 ^ (key >> 9);
73 d9 = d9 ^ (key & 0x1FF);
74 d9 = S9[d9] ^ d7;
75
76 return ((d7 << 9) | d9);
77}

79static uint32_t
80MISTY1_FO(BCMISTY1Context *ctx, uint32_t in, int idx)
81{

84

119static uint32_t bc_MISTY1_fl(BCMISTY1Context *ctx, uint32_t in, uint8_t idx)
120{
