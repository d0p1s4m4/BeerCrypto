34void
35bc_arcfour_init(BCArcfourContext *ctx, const uint8_t *key, size_t length)
36{
37 int i, j;
38 uint8_t c;
39
40 for (i = 0; i < 256; i++)
41 {
42 ctx->state[i] = i;
43 }
44
45 for (i = 0, j = 0; i < 256; i++)
46 {
47 j = (j + ctx->state[i] + key[i % length]) % 256;
48 c = ctx->state[i];
49 ctx->state[i] = ctx->state[j];
50 ctx->state[j] = c;
51 }
52
53 ctx->x = 0;
54 ctx->y = 0;
55}

57uint8_t *
58bc_arcfour_encrypt(BCArcfourContext *ctx, const uint8_t *in, size_t length, uint8_t *out)
59{
60 size_t i;
61 uint8_t c;
62
63 if (out == NULL)
64 {
65 return (NULL);
66 }
67
68 for (i = 0; i < length; i++)
69 {
70 ctx->x = (ctx->x + 1) % 256;
71 ctx->y = (ctx->y + ctx->state[ctx->x]) % 256;
72
73 /* swap */
74 c = ctx->state[ctx->x];
75 ctx->state[ctx->x] = ctx->state[ctx->y];
76 ctx->state[ctx->y] = c;
77
78 /* xor */
79 c = (ctx->state[ctx->x] ^ ctx->state[ctx->y]) % 256;
80 out[i] = in[i] ^ ctx->state[c];
81 }
82 return (out);
83}
